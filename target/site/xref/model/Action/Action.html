5   public class Action {
6   
7   	/*
8   	 * This method get the Object on the next case and the type of object which
9   	 * want to move on this sprite (Lorann, Monster, Spell) and will choose the
10  	 * appropriate action
11  	 */
12  
13  	private Kill kill;
14  
15  	public Action(ObjectComportement comp1, ObjectComportement comp2, ObjectComportement comp3, String type, int x,
16  			int y) {
17  		// If the next sprite is blocking for Lorann, a monster or either a
18  		// spell
19  		if (comp1 == ObjectComportement.BLOCKING || comp1 == ObjectComportement.POINTS && type == "monster1"
20  				|| comp1 == ObjectComportement.POINTS && type == "monster2"
21  				|| comp1 == ObjectComportement.POINTS && type == "monster3"
22  				|| comp1 == ObjectComportement.POINTS && type == "monster4"
23  				|| comp1 == ObjectComportement.POINTS && type == "Spell") {
24  			Block block = new Block(type);
25  		}
26  		// If there is the Ending door on the next sprite
27  		if (comp1 == ObjectComportement.MORTAL
28  				&& (type != "monster1" || type != "monster2" || type != "monster3" || type != "monster4")) {
29  			this.kill = new Kill(type);
30  		}
31  		// If there is a purse
32  		if (comp1 == ObjectComportement.POINTS && type == "Lorann") {
33  			Points points = new Points();
34  		}
35  		// If there is an energetic Bubble on the next Sprite
36  		if (comp1 == ObjectComportement.LINKED && comp2 == ObjectComportement.PENETRABLE) {
37  			ChangeDoor door = new ChangeDoor();
38  		}
39  		// If the object is penetrable
40  		if (comp2 == ObjectComportement.PENETRABLE && comp1 != ObjectComportement.MORTAL && type == "Lorann") {
41  			Movement move = new Movement(x, y, type);
42  		}
43  		if(comp1 == ObjectComportement.LEVELCHANGER && type =="Lorann"){
44  			ChangeLevel level = new ChangeLevel();
45  		}
46  		}
47  
48  	/*
49  	 * this method will choose the appropriate action to do
50  	 */
51  
52  	public Action(String NextType, String type) {
53  		if ((NextType == "monster1" || NextType == "monster2" || NextType == "monster3" || NextType == "monster4")
54  				&& type == "Lorann") {
55  			this.kill = new Kill(type);
56  		} else if (NextType == "Lorann" && (type == "monster1" || type == "monster2" || type == "monster3"
57  				|| type == "monster4")) {
58  			this.kill = new Kill(NextType);
59  		}
60  
61  		if ((NextType == "monster1" || NextType == "monster2" || NextType == "monster3" || NextType == "monster4")
62  				&& type == "Spell") {
63  			this.kill = new Kill(NextType);
64  		} else if (NextType == "Spell" && (NextType == "monster1" || NextType == "monster2" || NextType == "monster3"
65  				|| NextType == "monster4")) {
66  			this.kill = new Kill(type);
67  		}
68  
69  		if (NextType == "Spell" && type == "Lorann" || NextType == "Lorann" && type == "Spell") {
70  			RecoverSpell spell = new RecoverSpell();
71  		}
72  	}
